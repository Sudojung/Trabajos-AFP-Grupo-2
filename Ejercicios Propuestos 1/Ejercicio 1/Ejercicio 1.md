# Ejercicio 1:

Realizar un algoritmo que muestre por pantalla la tabla de multiplicar del 3 y del 9.

# Algoritmo:

INICIO

    Definir i, resultado1, resultado2 Como Entero
    
    Escribir "TABLA DEL 3 (Del 1 al 10)"
    
    Para i <- 1 Hasta 10 Hacer
        resultado1 <- 3 * i
        Escribir "3 x ", i, " = ", resultado1
    FinPara
    
    Escribir ""
    
    Escribir "TABLA DEL 9 (Del 1 al 10)"
    
    Para i <- 1 Hasta 10 Hacer
        resultado2 <- 9 * i
        Escribir "9 x ", i, " = ", resultado2
    FinPara

FIN
